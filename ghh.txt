# Get the resource path
var resource_path = "res://PlayerStats.tres"

# Load the resource
var player_stats = load(resource_path)

# Check if the resource loaded successfully
if player_stats:
	# Modify a property
	player_stats.health = 50
	player_stats.experience += 10

	# Save the resource with the new values
	var error = ResourceSaver.save(player_stats, resource_path)
	
	# Check for any save errors
	if error != OK:
		print("Error saving resource: ", error)
	else:
		print("Resource saved successfully!")
else:
	print("Error loading resource at: ", resource_path)
